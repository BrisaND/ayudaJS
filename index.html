Elemento DOM (HTML)


                    document.querySelector("tag") // <tag></tag> // devuelve eemento por etiqueta
                    
                    document.querySelector(clase) // class // devuelve elemennto por clase
                    
                    document.querySelector(#id) // devuelve elemento por id
                    
                    document.querySelectorAll(attr='valor_atributo') // src // devuelve un elemento del valor_atributo
                    
                    -------------------------------------------------------------------------------------------------------------
                  
                    document.querySelector('etiqueta.clase#id') // devuelve un arreglo de elementos que coinciden con el parametro
                  
                    -------------------------------------------------------------------------------------------------------------

                    document.querySelectorById('id') // devuelve un arreglo por id
                    
                    document.querySelectorByClassName('clase') // devuelve un arreglo por clase
                    
                    document.querySelectorTagName('etiqueta') // devuelve un arreglo por etiqueta
